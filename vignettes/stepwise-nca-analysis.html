#we need nominal time variable for some tasks.
NTAD <- c(0,0.25,0.5,1,2,4,5,7,9,12,24)

# or use metrumrg::snap
for(i in 1:nrow(Theoph)){
  time  <- Theoph$Time[[i]]
  delta <- abs(NTAD - time)
  best  <- min(delta)
  index <- match(best, delta)
  nom   <- NTAD[[index]]
  Theoph$NTAD[[i]] <- nom
}

Theoph1 <- Theoph %>% 
  # mutate(NTAD=metrumrg::snap(Time, NTAD)) %>%
  mutate(Subject=as.numeric(Subject)) %>%
  mutate(BLQ = ifelse(conc<0.20, 1, 0),
         LOQ = 0.20) %>%
  mutate(conc = ifelse(BLQ==1, 0, conc)) %>%
  as.data.frame() %>%
  arrange(Subject, Time)

#Calculate Cmax and Tmax

ctmax <- Theoph1 %>% calc.ctmax(
  by = 'Subject',
  timevar="Time",
  depvar="conc"
)

#Corrections of missing time and concentrations

## 3. Correct deviations
tc = Theoph1 %>%
  correct.loq(
    by='Subject',
    nomtimevar="NTAD",
    timevar="Time",
    depvar="conc",
    bloqvar="BLQ",
    loqvar="LOQ",
    loqrule=1
  ) %>%
  correct.time(
    by='Subject',
    nomtimevar="NTAD",
    timevar="Time",
    depvar="conc",
    th=th,
    reg="sd"
  ) %>%  
  correct.conc(
    by ='Subject',
    nomtimevar="NTAD",
    teval=12,
    th=th,
    reg="sd",
    ss="n",
  )
